function sumIntervals(intervals){
    var numbers = [];
    intervals.forEach(function (interval){
        for(var i=interval[0];i<interval[1];i++){
            if(numbers.indexOf(i)==-1)numbers.push(i);
        }
    });
    return numbers.length;
}
