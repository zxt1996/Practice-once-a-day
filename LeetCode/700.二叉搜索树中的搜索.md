# 700.二叉搜索树中的搜索
![](img/700.二叉搜索树中的搜索.png)  

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} val
 * @return {TreeNode}
 */
var searchBST = function(root, val) {
    if(!root){
        return null;
    }

    if(root.val == val){
        return root;
    }

    return searchBST(root.left, val) || searchBST(root.right, val)
};
```
