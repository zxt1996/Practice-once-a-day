# 122.买卖股票的最佳时机II
![](img/122.买卖股票的最佳时机II.png)  

```
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
    let res = 0;

    for(let i =1;i<prices.length;i++) {
        if(prices[i] > prices[i-1]) {
            res += prices[i] - prices[i-1];
        }
    }

    console.log(res);

    return res;
};
```
