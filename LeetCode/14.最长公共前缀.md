# 14.最长公共前缀

/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
    if(strs.length == 0 ){
        return '';
    }
    if(strs.length == 1){
        return strs[0];
    }
    let tempnum = strs[0].length,result=[];
    strs.reduce((pre,current)=>{
      if(tempnum > current.length){
        tempnum = current.length;
      }
    },tempnum);
    for(let i=0;i<tempnum;i++){
      let temp = strs[0].split('')[i];
      let j=0;
      while(j<strs.length){
        if(strs[j].split('')[i] != temp){
          if(result.length > 0){
            return result.join('');
          }else{
            return '';
          }
        }
        else if(j == strs.length-1){
          result.push(temp);
        }
        j++;
      }
    }
    console.log(result);
    if(result.length > 0){
      return result.join('');
    }else{
      return '';
    }
};
