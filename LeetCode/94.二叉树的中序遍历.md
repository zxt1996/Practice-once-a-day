# 94.二叉树的中序遍历
![](img/94.二叉树的中序遍历.png)  

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var inorderTraversal = function(root) {
    let temp = [];
    function mid(roots){
        if(roots!=null){
            mid(roots.left);
            temp.push(roots.val);
            mid(roots.right);
        }
    }
    mid(root);
    return temp;
};
```
