# 206.反转链表
![](img/206.反转链表.png)  

```
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    let pre = null;
    let temp = null;
     while(head){
        temp = head.next;
        head.next = pre;
        pre = head;
        head = temp;
     }
     return pre;
};
```
